GitHub API push pattern (mandatory — always follow this):

import urllib.request, json, base64

PAT = "[join patA + patB above]"
REPO = "ProfDrTan/research-methods"

# GET current SHA first — always
req = urllib.request.Request(
    f"https://api.github.com/repos/{REPO}/contents/FILENAME",
    headers={"Authorization": f"token {PAT}"}
)
with urllib.request.urlopen(req) as r:
    d = json.load(r)
sha = d['sha']

# PUT with content + sha
with open('/path/to/file', 'rb') as f:
    content = base64.b64encode(f.read()).decode()

payload = json.dumps({
    "message": "commit message here",
    "content": content,
    "sha": sha
}).encode()

req = urllib.request.Request(
    f"https://api.github.com/repos/{REPO}/contents/FILENAME",
    data=payload,
    headers={"Authorization": f"token {PAT}", "Content-Type": "application/json"},
    method="PUT"
)
with urllib.request.urlopen(req) as r:
    result = json.load(r)
